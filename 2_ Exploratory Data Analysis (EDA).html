import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
import numpy as np

# Loading the dataset
data = pd.read_csv('household_power_consumption.csv')

# Printing the column names to verify
print("Columns in the dataset:")
print(data.columns.tolist())

# Ensuring 'Datetime' column is in datetime format
data['Datetime'] = pd.to_datetime(data['Datetime'])

# Setting the 'Datetime' column as the index
data.set_index('Datetime', inplace=True)

# Replacing non-numeric values with NaN and convert columns to numeric
data.replace('?', np.nan, inplace=True)
data = data.apply(pd.to_numeric, errors='coerce')

# Handling missing values
data = data.fillna(method='ffill').fillna(method='bfill')

# Verifying that 'Global_active_power' exists and is numeric
if 'Global_active_power' in data.columns and pd.api.types.is_numeric_dtype(data['Global_active_power']):

    # Correlation matrix
    plt.figure(figsize=(10, 8))
    sns.heatmap(data.corr(), annot=True, fmt='.2f', cmap='coolwarm')
    plt.title('Correlation Matrix')
    plt.show()
else:
    print("Column 'Global_active_power' not found in the dataset or it is not numeric.")
